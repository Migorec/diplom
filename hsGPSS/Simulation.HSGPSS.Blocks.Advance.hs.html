1 {-# LANGUAGE FlexibleInstances #-}
    2 
    3 module Simulation.HSGPSS.Blocks.Advance where
    4 
    5 import Simulation.HSGPSS.Blocks
    6 
    7 data Pr = Pr Double 
    8 
    9 class AdvanceClass a where
   10     advance :: a -> BlockStateMonad
   11     
   12 instance AdvanceClass Double where
   13     advance m = addBlock $ AdvanceRange m 0
   14     
   15 instance AdvanceClass (Double, Double) where
   16     advance (m, r) = addBlock $ AdvanceRange m r
   17     
   18 instance AdvanceClass (Double, Double -> Double) where
   19     advance (m, f) = addBlock $ AdvanceFunc m f
   20     
   21 instance AdvanceClass Pr where
   22     advance (Pr p) = addBlock $ AdvanceParam (round p)
