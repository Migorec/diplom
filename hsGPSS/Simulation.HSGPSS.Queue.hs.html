    2 
    3 data SQueue = SQueue { currentContent :: Int,
    4                        maximumContent :: Int,
    5                        averageContent :: Double,
    6                        lastChangeTime :: Double
    7                      } deriving (Eq, Show)
    8                       
    9 initQueue = SQueue 0 0 0 0
   10 
   11 modify :: Double -> Int -> SQueue -> SQueue
   12 modify time inc (SQueue cc mc ac lt) = 
   13     if newCount < 0 
   14     then error "There are no enough entities in queue"
   15     else SQueue newCount 
   16                 (max mc newCount) 
   17                 ((ac * lt + fromIntegral cc * dt) / time) 
   18                 time
   19     where newCount = cc + inc
   20           dt = time - lt
